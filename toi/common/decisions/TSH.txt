TSH_category_shogun_succession_crisis = {
	
	#Succession Crisis!
	##Esentially, it's a timer that will count down for two months. At the end of that countdown, it will pick an heir based on influence. This to both force a choice in the matter, and to keep tempo.
	TSH_mission_pick_heir = {
		
		icon = generic_nationalism
		
		available = {
			has_country_flag = succession_crisis_starts
		}
		
		activation = {
			has_country_flag = succession_crisis_starts
		}
		
		visible = {
			TAG = TSH
			has_country_flag = succession_crisis_starts
			NOT = { has_country_flag = succession_crisis_ends }
		}
		
		days_mission_timeout = 30
		
		timeout_effect = {
			set_country_flag = succession_crisis_ends
			set_country_flag = iemochi_is_selected
			TSH = { country_event = { days = 5 id = shogunate_japan_l.3 } }
			
			if = {
				limit = { has_idea = TSH_succession_struggle }
				remove_ideas = TSH_succession_struggle
			}
		}
		
		complete_effect = {
			
		}
		
	}
}
