	TSH_mission_pick_heir = {
		
		icon = generic_nationalism
		
		allowed = {
			always = no
		}
		
		available = {
			hidden_trigger = {
				always = no
			}
		}
		
		days_mission_timeout = 30
		
		activation = {
			has_country_flag = succession_crisis_starts
		}
		
		visible = {
			TAG = TSH
			has_country_flag = succession_crisis_starts
			NOT = { has_country_flag = succession_crisis_ends }
		}
		
		timeout_effect = {
			set_country_flag = succession_crisis_ends
			set_country_flag = iemochi_is_selected

			if = {
				limit = { has_idea = TSH_succession_struggle }
				remove_ideas = TSH_succession_struggle
			}
		}
		
		complete_effect = {
			
		}
		
	}
