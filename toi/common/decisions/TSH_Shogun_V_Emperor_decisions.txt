Tokugawa_Shogun_Emperor_Influence_Category = {
	TSH_Suppress_National_Dissent = {
		allowed = { tag = TSH }

		available = {
			has_idea = TSH_Spirit_Unpopular_Shogunate
			NOT = { has_global_flag = JAP_Boshin_War_Ongoing }
			custom_trigger_tooltip = {
				tooltip = TSH_Any_National_Shogunate_Dissent_tt
				any_owned_state = {
					check_variable = { TSH.Meiji_Emperor_Influence@THIS > 15 }
				}
			}	
		}

		visible = {
			has_idea = TSH_Spirit_Unpopular_Shogunate
			any_owned_state = {
				check_variable = { TSH.Meiji_Emperor_Influence@THIS > 15 }
			}
		}

		days_re_enable = 365

		complete_effect = {
			custom_effect_tooltip = TSH_Suppress_National_Dissent_tt 
			hidden_effect = {
				every_owned_state = {
					limit = { check_variable = { TSH.Meiji_Emperor_Influence@THIS > 15 } }
					randomize_temp_variable = {
						var = TSH_Meiji_Influence_Change
						distribution = uniform
						min = -9
						max = -5
					}
					round_temp_variable = TSH_Meiji_Influence_Change
					TSH_Meiji_Influence_Modify = yes
				}
			}
		}
	}
	TSH_Meiji_Spread_Influence_Example = {
		allowed = { always = no }

		available = {
			# Which countries are allowed to clicky?
			OR = {
				has_country_flag = TSH_Meiji_Influence_Outside_Influencer
				is_subject_of = TSH
			}
			# Do they own a neighboring state?
			OR = {
				owns_state = 1
			}
			# Is that state still a valid target?
			1 = {
				custom_trigger_tooltip = {
					tooltip = TSH_Meiji_Spread_Influence_Available_tt
					check_variable = { TSH.Meiji_Emperor_Influence@THIS > 0 }
					check_variable = { TSH.Meiji_Emperor_Influence@THIS < 100 }
				}
			}
		}

		visible = {
			OR = {
				has_country_flag = TSH_Meiji_Influence_Outside_Influencer
				is_subject_of = TSH
			}
			OR = {
				owns_state = 1
			}
			1 = {
				custom_trigger_tooltip = {
					tooltip = TSH_Meiji_Spread_Influence_Available_tt
					check_variable = { TSH.Meiji_Emperor_Influence@THIS > 0 }
					check_variable = { TSH.Meiji_Emperor_Influence@THIS < 100 }
				}
			}
		}

		highlight_states = { state = 1 }

		cost = 25
		custom_cost_trigger = { command_power > 24.9 }
		custom_cost_text = command_power_more_than_24

		days_re_enable = 30

		complete_effect = {
			1 = {
				set_temp_variable = { TSH_Meiji_Influence_Change = 2 }
				TSH_Meiji_Influence_Modify = yes
			}
			hidden_effect = {
				add_command_power = -25
			}
		}
	}
}
