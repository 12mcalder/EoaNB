# Enable your country to have crisises by adding them in the allowed of this crisis
# (should be stored in eoanb_generic_decision_categories.txt)

EOANBSYS_pandemic_decision_category = {
	EOANBSYS_pandemic_progress_counter = {
		icon = generic_civil_support

		allowed = { always = no } # Don't Touch

		activation = { always = no } # Don't Touch

		available = {
			has_pandemic = no
		}

		days_mission_timeout = 14

		timeout_effect = {
			EOANBSYS_pandemic_progress_add_half = yes
			if = {
				limit = { check_variable = { EOANBSYS_pandemic_progress_var < 100 } }
				EOANBSYS_pandemic_wave = yes
				activate_mission = EOANBSYS_pandemic_progress_counter_14
			}
			else = {
				EOANBSYS_pandemic_clear = yes
			}
		}

		complete_effect = {
			EOANBSYS_pandemic_clear = yes
		}
	}
}
