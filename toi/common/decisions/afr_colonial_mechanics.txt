						OR = {
							has_idea = AFR_illegal_transatlantic

					}

			}
		}

		visible = {
			OR = {	
				has_idea = slavery

		}

		ai_will_do = {
			factor = 1
		}

		complete_effect = {
			remove_ideas = AFR_transatlantic_consumer
		}
	}

	}
