			ENG = { 
				is_neighbor_of = ROOT
			}
			any_country = {
				AND = {
					is_subject_of = ENG
					is_neighbor_of = ROOT
				}
			}
		}
		if = {
			limit = {
				ROOT = {
					is_subject_of = OTO
				}
			}
			OTO = {
				has_completed_focus = OTO_bank_of_england_partnership
			}
		}
    }
    visible = {
		country_exists = ENG
		AND = {
			TAG = ETH
			has_country_flag = ETH_open_trading_post
		}
    }
	
	fire_only_once = yes

    cost = 0
	
	
	complete_effect = {
		ENG = { country_event = { days = 1 id = ethiopia_trade.1 } }
		hidden_effect = {	
			set_country_flag = ETH_ENG_Merchants

			reverse_add_opinion_modifier = { target = ENG modifier = eoanb_om_africa_trade_focus }
