	RYU_build_farm = {

		available = {
			if = {
				limit = {
					capital_scope = { cotton_farm > 7 }
				}
				custom_trigger_tooltip = {
					tooltip = RYU_prosperity_mechanic_greater_than_3
						check_variable = {
						ryu_prosperity_mechanic > 2
					}
				}
				has_tech = mechanized_farming
			}
			else = {
				custom_trigger_tooltip = {
					tooltip = RYU_prosperity_mechanic_greater_than_3

			TAG = RYU
		}

			political_power_cost = 0.5
