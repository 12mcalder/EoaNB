shared_focus = {
  id = NA_B_recon_industrialize_1
  icon = GFX_goal_focus_industrial_revolution
  cost = 105

  prerequisite = { focus = NA_B_recon_infrastructure_effort_1 }
  relative_position_id = NA_B_recon_infrastructure_effort_1
  x = 2
  y = 1

  available_if_capitulated = no
  available = {
    num_of_factories > 1
    has_political_power > 0
    has_stability > 0.2
  }
  ai_will_do = { base = 20 }
  bypass = {
    OR = {
      has_idea = victorian_era
      has_idea = machine_era
      has_idea = oil_era
      has_idea = aviation_era
      has_idea = modern_era
      has_idea = atomic_era
      has_idea = information_era
      has_idea = digital_era
      has_idea = post_digital_era
    }
  }

  completion_reward = {
    add_political_power = -100
    add_building_construction = {
      type = industrial_complex
      level = 1
      instant_build = yes
    }
    if = {
      has_idea = pre_industrial_era
      swap_ideas = {
        remove_idea = pre_industrial_era
        add_idea = victorian_era
      }
    }
  }
}
