d_add_culture={ ##add_culture ID PERCENTAGE STATE
	if={
		limit={
			check_variable= {args^2 > 0}
		}
		args^2={
			set_temp_variable = {type_temp = args^0}
			set_temp_variable = {percentage_temp = args^1}
		

			pops_increase_culture = yes
		}
	}
	else={
		ROOT={
			set_temp_variable = {type_temp = args^0}
			set_temp_variable = {percentage_temp = args^1}
		

			pops_increase_culture = yes
		}
	}
}
