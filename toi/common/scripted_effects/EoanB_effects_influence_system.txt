eoanb_influence_calc = {
	every_country = {
	    clear_variable = pp_scaling_influence
	    if = {
	        limit = { check_variable = { political_power > 1 } }
		    set_variable = { pp_scaling_influence = political_power }
	    	divide_variable = { pp_scaling_influence = 100 }
	    }
        else = {
		    set_variable = { pp_scaling_influence = 1 }
		}		
	}
}
