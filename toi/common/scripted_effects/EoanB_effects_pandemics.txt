	EOANBSYS_pandemic_setup = {
		ROOT = {
			add_ideas = EOANBSYS_spirit_pandemic
			set_variable = { EOANBSYS_pandemic_progress_var = 0 }
			activate_mission = EOANBSYS_pandemic_progress_counter
		}
	}

	EOANBSYS_pandemic_clear = {
		ROOT = {
			remove_ideas = EOANBSYS_spirit_pandemic
			clear_variable = EOANBSYS_pandemic_progress_var
			increase_ps_by_one = yes
			hidden_effect = {
				clear_variable = EOANBSYS_pandemic_progress_var
				remove_mission = EOANBSYS_pandemic_progress_counter
			}
		}
	}

	EOANBSYS_pandemic_wave = {
		ROOT = {
			if = {
				limit = { has_country_flag = has_country_flag = EOANBSYS_pandemics_cholera_flag }
				set_temp_variable = { EOANBSYS_pandemic_death_temp = EOANBSYS_pandemics_cholera_constant }
				every_owned_state = {
					random_list = {
						9 = {
							randomize_temp_variable = {
								var = EOANBSYS_pandemic_state_death_temp
								min = 0
								max = EOANBSYS_pandemic_death_temp
							}
							set_temp_variable = { EOANBSYS_pandemic_state_population_temp = state_population }
							multiply_temp_variable = { EOANBSYS_pandemic_state_population_temp = EOANBSYS_pandemic_state_death_temp }

						}
						1 = {
							modifier = {
								add = 1
							}
						}
					}
				}
			}
