recalc_rus_caucasian_variables = {
    if = {
	    limit = {
		    check_variable = { RUS_var_caucasus_unrest > 0.70 }
		}
		every_owned_state = {
		    limit = { has_state_flag = RUS_caucasian_medium_unrest_flag }
	    	remove_province_modifier = {
			    static_modifiers = { rus_caucasian_medium_unrest_prov_modifier }
                province = {
                    all_provinces = yes
                }				
			}
            add_province_modifier = {
                static_modifiers = { rus_caucasian_high_unrest_prov_modifier }
                province = {
                    all_provinces = yes
                }
            }
			clr_state_flag = RUS_caucasian_medium_unrest_flag
            set_state_flag = RUS_caucasian_high_unrest_flag			
		}
		else_if = {
        	limit = {
        		check_variable = { RUS_var_caucasus_unrest < 0.30 }
	        }
	     	every_owned_state = {
	    	    limit = { has_state_flag = RUS_caucasian_medium_unrest_flag }
	        	remove_province_modifier = {
		   	        static_modifiers = { rus_caucasian_medium_unrest_prov_modifier }
                    province = {
                        all_provinces = yes
                    }				
			    }
                add_province_modifier = {
                    static_modifiers = { rus_caucasian_low_unrest_prov_modifier }
                    province = {
                        all_provinces = yes
                    }
                }
				clr_state_flag = RUS_caucasian_medium_unrest_flag
                set_state_flag = RUS_caucasian_low_unrest_flag 				
		    }
	    }
		else = {
	     	every_owned_state = {
			    OR = {
	    	        limit = { has_state_flag = RUS_caucasian_low_unrest_flag }
					limit = { has_state_flag = RUS_caucasian_high_unrest_flag }
				}
	        	remove_province_modifier = {
		   	        static_modifiers = { rus_caucasian_high_unrest_prov_modifier }
					static_modifiers = { rus_caucasian_low_unrest_prov_modifier }
                    province = {
                        all_provinces = yes
                    }				
			    }
                add_province_modifier = {
                    static_modifiers = { rus_caucasian_medium_unrest_prov_modifier }
                    province = {
                        all_provinces = yes
                    }
                }
				clr_state_flag = RUS_caucasian_low_unrest_flag
				clr_state_flag = RUS_caucasian_high_unrest_flag
                set_state_flag = RUS_caucasian_medium_unrest_flag				
		    }		
		}
	}
}
