defined_text = {
	name = GetMHCNavyNasImageAUS
	
	text = {
		trigger = {
			has_variable = local_mhc_generic_navy_advisors_array^0
			is_in_array = {
				array = local_mhc_generic_navy_advisors_array
				value = mhc_nas_value
			}
		}
		localization_key = "[GetMHCNavyNasImageGeneric]"
	}
	text = {
		trigger = {
			check_variable = { mhc_nas_value = 1 }
		}
		localization_key = "GFX_idea_AUS_theorist_Archduke_Maximillian"
	}
	text = {
		trigger = {
			check_variable = { mhc_nas_value = 2 }
		}
		localization_key = "GFX_idea_AUS_advisor_Ludwig_von_Fautz"
	}
	text = {
		trigger = {
			check_variable = { mhc_nas_value = 3 }
		}
		localization_key = "GFX_idea_AUS_vic_Littrow"
	}
	text = {
		trigger = {
			check_variable = { mhc_nas_value = 4 }
		}
		localization_key = "GFX_idea_AUS_vic_Maximilian_Daublebsky_von_Sterneck"
	}
	###Navy Nas Replacement Marker###
	
	text = {
		trigger = {
			check_variable = { mhc_nas_value = 0 }
		}
		localization_key = "GFX_MHC_null"
	}
}
