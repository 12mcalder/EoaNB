				compare = greater_than_or_equals
			}
		}
		localization_key = "GetSuezConstructionAvailibleComplete"
	}
	text = {

				}
			}
		}
		localization_key = "GetSuezConstructionAvailibleYes"
	}
	text = {

			NOT={EGY={has_country_flag = suez_constructor}}
