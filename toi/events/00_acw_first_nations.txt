country_event = {
	id = acw_first_nations_trigger.1 
	hidden = yes

	is_triggered_only = yes
	fire_only_once = yes

    immediate = {
		hidden_effect = {
			if = {
				limit = {
					AND = {
						USA = { has_country_flag = USA_American_Civil_War }
						CSA = { has_country_flag = CSA_American_Civil_War }
					}
				}
				every_country = {
					limit = {
						OR = {
							TAG = SHO
							TAG = NEZ
							TAG = BFT
							TAG = CRW
							TAG = SIX
							TAG = CHY
							TAG = COM
							TAG = APA
							TAG = NAV
							TAG = SEM
							TAG = OKL
						}
					}
					country_event = { id = acw_first_nations.1 days = 7 }
				}
			}
		}
	}
}
