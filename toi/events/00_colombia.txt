#Supreme Director of War declares sovereign state a separate nation, causing civil war to break out
country_event = {
    id = colombia.3
    title = colombia.3.t
    desc = colombia.3.d
	picture = GFX_colombia_civil_war

    fire_only_once = yes

    trigger = {
        tag = COL
        date > 1860.5.8
    }
    option = {
        name = colombia.3.a
        hidden_effect = {
            start_civil_war = {
                ruling_party = traditional_conservatism
                ideology = centrism
				size = 0.86
				capital = 486
                keep_unit_leaders = {
                    687271 687282 687293
                }
            }
			every_country = {
                limit = {
                    original_tag = COL
                    has_government = centrism
                }
                set_cosmetic_tag = COL_cauca
            }
            486 = {
                controller = {
                    COL = {
                        every_unit_leader = {
                            limit = {
                                has_id = 687304
                            }
                            set_nationality = PREV.PREV
                        }
                    }
                    every_unit_leader = {
                        limit = {
                            NOT = {
                                has_id = 687304
                            }
                        }
                        retire = yes
                    }
                }
            }
        }
	}
}
#Granadine Confederation becomes United states of Colombia		
country_event = {
    id = colombia.4
    title = colombia.4.t
    desc = colombia.4.d

    hidden = yes

    fire_only_once = yes

    trigger = {
        306 = {
            owner = {
                has_civil_war = no
                has_government = centrism
            }
        }
    }

    immediate = {
        every_country = {
            limit = {
                original_tag = COL
                has_government = centrism
            }
            change_tag_from = COL
        }
        COL = {
            set_cosmetic_tag = COL_united
        }
    }
}
