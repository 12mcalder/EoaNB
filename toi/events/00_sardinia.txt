	id = sardinia.27
	title = sardinia.27.t
	desc = sardinia.27.d
	fire_only_once = yes
	is_triggered_only = yes
	trigger = {
		OR = {
			if = {
				limit = {
					FRA = {
						has_country_flag = FRA_supports_PIE_risorgimento
					}
				}
				#has_completed_focus = PIE_Recruit_Nationalists
				has_completed_focus = PIE_French_Aid
				has_completed_focus = PIE_Start_Border_Operations
			}
			has_completed_focus = PIE_Provoke_Austria
		}
		NOT = {
			has_country_flag = is_literally_italy
		}
	}
	option = {
		name = sardinia.27.a
		activate_mission = ITA_Mission_AUS_PIE_Peace_Treaty_1860
		AUS = {
			activate_mission = ITA_Mission_AUS_PIE_Peace_Treaty_1860
		}
		custom_effect_tooltip = PIE_LVN_War
		hidden_effect = {
			set_country_flag = PIE_2I_war_active
		}
		if = {
			limit = {
				FRA = {
					has_country_flag = FRA_supports_PIE_risorgimento
				}
			}
			hidden_effect = {
				complete_national_focus = PIE_The_War_Starts
			}
			hidden_effect = {
				complete_national_focus = PIE_French_Army_Arrives
			}
			hidden_effect = {
				complete_national_focus = PIE_Occupy_Central_Italy
			}
		}
		if = {
			limit = {
				is_ai = yes
			}
			add_ideas = {
				ITA_AI_Boost_War
			}
		}
		declare_war_on = {
			target = LVN
			type = annex_everything
		}
		#Russia intervenes if there was no conclusive end to the Zweikaisertreffen in Württemberg
		if = {
			limit = {
				WUR = {
					NOT = {
						has_country_flag = WUR_zweikaisertreffen_success
					}
				}
				PIE = {
					NOT = {
						has_completed_focus = PIE_Ensure_Russias_Neutrality
					}
				}
			}
			RUS = {
				country_event = {
					id = wurttem.52
				}
			}
		}
	}
}
