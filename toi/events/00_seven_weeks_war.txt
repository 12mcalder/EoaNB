#Biarritz Meeting
country_event = {
	id = sevenweekswar.1
	title = sevenweekswar.1.t
	desc = sevenweekswar.1.d
	picture = GFX_report_event_hitler_handshake
	
	fire_only_once = yes
	
	trigger = {
		OR = {
			AND = {
				tag = PRS
				PRS = { has_country_flag = PRS_gastein_convetion }
				PRS = { has_war = no }
				date > 1865.8.1
			}
			AND = {
				tag = PRS
				PRS = { NOT = { has_country_flag = PRS_gastein_convetion } }
				PRS = { has_war = no }
				date > 1867.1.1
			}
		}
	}
	
	mean_time_to_happen = { days = 20 }
	
	option = {
		name = sevenweekswar.1.a
		ai_chance = { factor = 100 }
		add_political_power = 50
		add_opinion_modifier = { target = FRA modifier = PRS_FRA_ARRANGEMENTS_REL }
		reverse_add_opinion_modifier = { target = FRA modifier = PRS_FRA_ARRANGEMENTS_REL }
		hidden_effect = {
			PRS = { country_event = { days = 150 id = sevenweekswar.2 } }
			AUS = { country_event = { days = 150 id = sevenweekswar.3 } }
			every_other_country = {
				limit = {
					has_idea = GER_general_idea_ger_confed
					NOT = { tag = PRS }
					NOT = { tag = AUS }
				}
				country_event = { days = 150 id = sevenweekswar.4 } 
			}
		}
	}
}
