		fire_only_once = yes
		
		is_triggered_only = yes
		
		#The following options can only have either of them appear in-game. Each button is respective to each heir, and will fire the subsequent events that put them as leaders (exception for Iemochi who becomes HoS instead)

		option = {
			name = shogunate_japan_l.3.a
			
			ai_chance = { factor = 1 }
			
			trigger = {
				has_country_flag = iemochi_is_selected
			}
			
			hidden_effect = {
				TSH = { country_event = { days = 5 id = shogunate_japan_l.4 } }
			}
			
			custom_effect_tooltip = TSH_iemochi_is_selected_tt
		}
		
		#One man dies, another rises... (Yoshinobu)

			trigger = {
				has_country_flag = yoshinobu_is_selected
			}
			
			custom_effect_tooltip = TSH_yoshinobu_is_selected_tt
			
		}
