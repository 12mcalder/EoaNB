#Mexican Civil War erupts
country_event = {
		id = mexico.6
		title = mexico.6.t
		desc = mexico.6.d
		picture = GFX_report_event_mex_civil_war
		
		fire_only_once = yes

		is_triggered_only = yes
		
		option = {
			name = mexico.6.a
			custom_effect_tooltip = MEX_1857_Civilwar
			hidden_effect = {
				start_civil_war = { 
					ideology = centrism
					size = 0.5
					states = { 1124 477 478 479 480 485 1112 }
					capital = 1124
					keep_unit_leaders = {
						18626 186211
					}
				}

				random_other_country = {
					limit = {
						original_tag = MEX
						has_government = centrism
					}
					save_event_target_as = mexican_liberals
					load_oob = "Mex_civil_war_liberals"
					country_event = { days = 2 id = mexico_gaa.3 }
				}
				MEX = {
					every_unit_leader = {
						LIMIT = {
							OR = {
								has_id = 18625 
								has_id = 186210
							}
						}
						set_nationality = event_target:mexican_liberals
					}
				}
				load_oob = "Mex_civil_war_conservatives"
				MEX = { country_event = { days = 2 id = mexico_gaa.2 } }
				news_event = news.4
			}
		}
		option = {
			name = mexico.6.b
			custom_effect_tooltip = MEX_1857_Civilwar
			set_capital = 1124
			set_politics = { ruling_party = centrism elections_allowed = yes }
			hidden_effect = {
				increase_ps_by_one = yes
				start_civil_war = { 
					ideology = traditional_conservatism
					size = 0.50
					capital = 277
					states = { 277 474 475 476 481 482 483 484 1113 1114 1125 }
					keep_unit_leaders = { 18625 186210 }
				}
				load_oob = "Mex_civil_war_liberals"
				MEX = { country_event = { days = 2 id = mexico_gaa.3 } }

				random_other_country = {
					limit = {
						original_tag = MEX
						has_government = traditional_conservatism
					}
					save_event_target_as = mexican_conservatives
					load_oob = "Mex_civil_war_conservatives"
					country_event = { days = 2 id = mexico_gaa.2 }
				}
				MEX = {
					every_unit_leader = {
						LIMIT = {
							OR = {
								has_id = 18626 
								has_id = 186211
							}
						}
						set_nationality = event_target:mexican_conservatives
					}
				}
				news_event = news.4
			}
		}
}
