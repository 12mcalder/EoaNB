			limit = {
				PRS = { 
					has_manpower > 13200

				}
			}
			load_oob = PRS_roon_reform_1

			add_political_power = -25
			add_stability = -0.01

				ideology = centrism
				popularity = 0.01
			}
			hidden_effect = { PRS = { country_event = { days = 180 id = prussiareformtroops.2 } } }
		}
		else_if = {
			limit = {
				PRS = { 
					OR = {
						AND = {	
							has_active_mission = PRS_mission_parliamentary_reform_bill
